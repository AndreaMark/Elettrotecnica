for R2 = [1:100]

1/R1+1/(2*R1)+1/R2 -(1/(2*R1)+1/R2)
-(1/(2*R1)+1/R2) 1/R1+1/(2*R1)+1/R2]

C1 = cond(An)

% Matrice dalle Correnti di Maglia

Ac = [...
R1+R2+R1 R2
R2 R2+2+R1]

C2 = cond(Ac)
end
end
